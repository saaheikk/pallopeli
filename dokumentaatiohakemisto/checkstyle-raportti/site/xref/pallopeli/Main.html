35      public static void startProgram() {
36          Game g = new Game(30);
37          
38          UserInterface ui = new UserInterface(g);
39          SwingUtilities.invokeLater(ui); // Causes ui.run() to be executed asynchronously on the AWT event dispatching thread.
40          
41          while (ui.getPainitingCanvas() == null) {
42              try {
43                  Thread.sleep(100);
44              } catch (InterruptedException ex) {
45                  System.out.println("Game view still under contruction.");
46              }
47          }  
48          while (ui.getSidebar() == null) {
49              try {
50                  Thread.sleep(100);
51              } catch (InterruptedException ex) {
52                  System.out.println("Game view still under contruction.");
53              }
54          }    
55          g.setPaintingCanvas(ui.getPainitingCanvas());
56          g.setSidebar(ui.getSidebar());
57          g.start(); // Starts the Timer, causing it to start sending action events to its listeners.        
58      }
