1   package pallopeli.gui;

3   import pallopeli.graphics.Updateable;
4   import pallopeli.graphics.PaintingCanvas;
5   import java.awt.Component;
6   import java.awt.Container;
7   import java.awt.Dimension;
8   import javax.swing.JFrame;
9   import javax.swing.WindowConstants;
10  import pallopeli.logic.Game;
11  
12  /**
13   * UserInterface contains JFrame that further contains the graphical user interface for this game.
14   * @author saara
15   */
16  
17  public class UserInterface implements Runnable {
18      private JFrame frame;
19      private Game game;
20      private int sizeOfObjects;    
21      private Updateable paintingCanvas;
22  
23      public UserInterface(Game game) {
24          this.game = game;
25          this.sizeOfObjects = game.getSizeOfObjects();
26      }
27      
28      /**
29       * Method inherited from Runnable; When an object implementing interface Runnable is used to create a thread, starting the thread causes the object's run method to be called in that separately executing thread.
30       */
31      @Override
32      public void run() {
33          frame = new JFrame("Pallopeli");
34          frame.setPreferredSize(new Dimension(300, 322));
35          frame.setDefaultCloseOperation(WindowConstants.EXIT_ON_CLOSE);
36  
37          createComponents(frame.getContentPane());
38  
39          frame.pack();
40          frame.setVisible(true);
41      }
42      
43      // This method will later add other ui-components to Container!
44      public void createComponents(Container container) {
45          // Huom! Luo ensin piirtoalusta jonka lisäät container-olioon
46          // Luo vasta tämän jälkeen näppäimistönkuuntelija, jonka lisäät frame-oliolle
47          this.paintingCanvas = new PaintingCanvas(this.game);
48          container.add((Component) paintingCanvas, "Center");
49  
50      }
51  
52      // getters and setters
53      
54      public JFrame getFrame() {
55          return frame;
56      }
57      public Updateable getUpdateable() {
58          return this.paintingCanvas;
59      }
60  }
