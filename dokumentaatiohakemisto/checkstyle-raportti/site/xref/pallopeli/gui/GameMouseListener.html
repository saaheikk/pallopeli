6   package pallopeli.gui;
7   
8   import java.awt.event.MouseEvent;
9   import java.awt.event.MouseListener;
10  import pallopeli.logic.Game;
11  import pallopeli.objects.Board;
12  import pallopeli.objects.Piece;
13  
14  /**
15   * CustomMouseListener listens to mouse clicks and sets Game into building mode (if it's not in building mode already) .
16   * @author saara
17   */
18  public class GameMouseListener implements MouseListener {
19      private Game game;
20      /**
21       * Constructor for a new GameMouseListener attached to given Game.
22       * @param game 
23       */
24      public GameMouseListener(Game game) {
25          this.game = game;
26      }
27  
28      @Override
29      public void mouseClicked(MouseEvent e) {
30          if (!this.game.isBuilding()) {
31              boolean startIdentified = this.game.getWallbuilder().resetStart(e.getPoint(), this.game.getDirection());
32              this.game.setBuilding(startIdentified);
33          }      
34      }
35  
36      @Override
37      public void mousePressed(MouseEvent e) {
38      }
39  
40      @Override
41      public void mouseReleased(MouseEvent e) {
42      }
43  
44      @Override
45      public void mouseEntered(MouseEvent e) {
46      }
47  
48      @Override
49      public void mouseExited(MouseEvent e) {
50      }
51      
52  }
