1   package pallopeli.objects;
2   
3   import java.awt.Point;
4   import java.awt.geom.Line2D;
5   import java.awt.geom.Point2D;
6   import pallopeli.CompassDirection;
7   
8   public class BorderLine extends Border {
9   
10      /**
11       * Constructor for a new Border that can de described as Line2D.
12       * @param piece
13       * @param directionOfEdgeToGuard
14       * @param radiusOfBall 
15       */
16      public BorderLine(Piece piece, CompassDirection directionOfEdgeToGuard, int radiusOfBall) {
17          super(piece, directionOfEdgeToGuard, radiusOfBall);
18  
19          Point firstCorner = piece.getCornerPoint(directionOfEdgeToGuard.getFirstSideOfMainDirection());
20          Point secondCorner = piece.getCornerPoint(directionOfEdgeToGuard.getSecondSideOfMainDirection());
21  
22          int firstX = firstCorner.x + directionOfEdgeToGuard.getX() * radiusOfBall;
23          int firstY = firstCorner.y + directionOfEdgeToGuard.getY() * radiusOfBall;      
24          int secondX = secondCorner.x + directionOfEdgeToGuard.getX() * radiusOfBall;
25          int secondY = secondCorner.y + directionOfEdgeToGuard.getY() * radiusOfBall;
26          this.borderShape = new Line2D.Float(firstX, firstY, secondX, secondY);
27  
28      }
29      
30      @Override
31      public Line2D getBorderShape() {
32          return (Line2D) this.borderShape;
33      }
34      /**
35       * Helper method to extend BorderLine to cover the corners of Piece.
36       * Used because I don't have time to implement BorderArc.
37       * Game works quite well without BorderArcs, however.
38       */
39      public void extendLine() {
40          Point2D firstEnd = this.getBorderShape().getP1();
41          Point2D secondEnd = this.getBorderShape().getP2();
42          int coefficientX = Math.abs(this.directionOfEdgeToGuard.getY());
43          int coefficientY = Math.abs(this.directionOfEdgeToGuard.getX());
44          int newFirstEndX = (int) firstEnd.getX() - coefficientX * this.radiusOfBall;
45          int newFirstEndY = (int) firstEnd.getY() - coefficientY * this.radiusOfBall;
46          int newSecondEndX = (int) secondEnd.getX() + coefficientX * this.radiusOfBall;
47          int newSecondEndY = (int) secondEnd.getY() + coefficientY * this.radiusOfBall;
48          this.borderShape = new Line2D.Float(newFirstEndX, newFirstEndY, newSecondEndX, newSecondEndY);
49      }
50       
51      /**
52       * Method used to eliminate copy paste in CollisionDetector.
53       * @return Integer to indicate which direction Ball must bounce.
54       */
55      public int getBouncingDirection() {
56          if (this.directionOfEdgeToGuard == CompassDirection.NORTH || this.directionOfEdgeToGuard == CompassDirection.WEST) {
57              return -1;
58          } else if (this.directionOfEdgeToGuard == CompassDirection.SOUTH || this.directionOfEdgeToGuard == CompassDirection.EAST) {
59              return 1;
60          }
61          return 0;
62      }
63      /**
64       * Returns the x-coordinate of vertical BorderLine and y-coordinate of horizontal BorderLine.
65       * @return Coordinate as integer.
66       */
67      public int getLineCoordinate() {
68          Line2D line = (Line2D) this.borderShape;
69          int coefX = Math.abs(this.directionOfEdgeToGuard.getX());
70          int coefY = Math.abs(this.directionOfEdgeToGuard.getY());        
71          return (int) (coefX * line.getX1() + coefY * line.getY1());
72      }
73  
74  }
