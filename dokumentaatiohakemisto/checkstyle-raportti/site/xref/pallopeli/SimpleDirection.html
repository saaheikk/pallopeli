3   /**
4    * SimpleDirection is mainly used to make Ball bounce (three different ways to reset speed).
5    * @author saara
6    */
7   
8   public enum SimpleDirection {
9       
10      HORIZONTAL(CompassDirection.WEST, CompassDirection.EAST, CompassDirection.NORTH, CompassDirection.SOUTH), 
11      VERTICAL(CompassDirection.NORTH, CompassDirection.SOUTH, CompassDirection.WEST, CompassDirection.EAST),
12      DIAGONAL(null, null, null, null);
13      
14      CompassDirection first;
15      CompassDirection second;
16      CompassDirection firstPerpendicular;
17      CompassDirection secondPerpendicular;
18      
19      SimpleDirection(CompassDirection first, CompassDirection second, CompassDirection firstPerpendicular, CompassDirection secondPerpendicular) {
20          this.first = first;
21          this.second = second;
22          this.firstPerpendicular = firstPerpendicular;
23          this.secondPerpendicular = secondPerpendicular;
24      }
25      /**
26       * Method used to eliminate copy paste in Wallbuilder.
27       * @param i Closer / further.
28       * @return CompassDirection closer to / further from origin.
29       */
30  
31      public CompassDirection getAssociatedCompassDirection(int i) {
32          if (i == 1) {
33              return first;
34          }
35          if (i == 2) {
36              return second;
37          }
38          return null;
39      }
40      /**
41       * Method used to eliminate copy paste in Wallbuilder.
42       * @param i Closer / further.
43       * @return Perpendicular CompassDirection closer to / further from origin.
44       */    
45      public CompassDirection getPerpendicularCompassDirection(int i) {
46          if (i == 1) {
47              return firstPerpendicular;
48          }
49          if (i == 2) {
50              return secondPerpendicular;
51          }
52          return null;
53      }    
54  
55      
56  }
