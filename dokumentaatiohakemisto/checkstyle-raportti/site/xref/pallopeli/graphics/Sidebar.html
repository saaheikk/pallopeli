6   package pallopeli.graphics;
7   
8   import java.awt.Color;
9   import java.awt.Container;
10  import java.awt.Dimension;
11  import java.util.ArrayList;
12  import javax.swing.BoxLayout;
13  import javax.swing.JLabel;
14  import javax.swing.JPanel;
15  import javax.swing.JTextField;
16  import pallopeli.logic.Game;
17  
18  /**
19   * Sidebar represents the info bar on the side of actual game board.
20   * Sidebar is updated during the game.
21   * @author saara
22   */
23  public class Sidebar extends JPanel implements Updateable {
24      private Game game;
25      
26      private JLabel instructions;
27      private ArrayList<JLabel> instructionsContent;
28      private JLabel lives;
29      private JLabel direction;
30      private JLabel status;
31      private JLabel statusContent;
32      
33      public Sidebar(Game game) {
34          super.setBackground(Color.WHITE);
35          this.game = game;
36          this.createContent(this);        
37      }  
38      /**
39       * Creates the content of Sidebar and adds it to Container.
40       * @param container Container (of JFrame).
41       */   
42      public void createContent(Container container) {
43          this.setLayout(new BoxLayout(container, BoxLayout.Y_AXIS));
44  
45          this.instructions = new JLabel();
46          this.instructionsContent = new ArrayList<>();
47          this.lives = new JLabel();
48          this.direction = new JLabel();
49          this.status = new JLabel();  
50          this.statusContent = new JLabel();
51          
52          container.add(this.instructions);
53          this.addInstructionsContent(container);
54          container.add(this.lives);
55          container.add(this.direction);
56          container.add(this.status);   
57          container.add(this.statusContent);   
58          this.setPreferredSize(new Dimension(180, 322));
59          this.setMaximumSize(this.getPreferredSize()); 
60          this.setMinimumSize(this.getPreferredSize());
61      }
62      
63      
64              
65  
66      @Override
67      public void update() {
68          this.instructions.setText("INSTRUCTIONS:");
69          this.lives.setText("LIVES: " + game.getLives());
70          this.direction.setText("DIRECTION: " + game.getDirection() + "\n");
71          this.status.setText("GAME STATUS: ");
72          this.statusContent.setText(this.statusOfGame());
73      }
74      /**
75       * Returns the current status of Game.
76       * @return Status as String to be inserted into JLabel.
77       */
78      
79      public String statusOfGame() {
80          String status = "";
81          if (game.continues()) {
82              status += "Game continues.";
83          } else {
84              if (game.win()) {
85                  status += "Game over! You win!";
86              } else {
87                  status += "Game over! You lost!";
88              }
89          }
90          return status;
91      }
92      /**
93       * Adds instructions as JLabels.
94       * @param container Container (of JFrame).
95       */
96      
97      public void addInstructionsContent(Container container) {
98          JLabel firstRow = new JLabel("Turn areas into wall by");
99          container.add(firstRow);
100         JLabel secondRow = new JLabel("clicking pieces. You need");
101         container.add(secondRow);
102         JLabel thirdRow = new JLabel("to get 90% of board covered");
103         container.add(thirdRow);        
104         JLabel fourthRow = new JLabel("to win the game. Change");
105         container.add(fourthRow);        
106         JLabel fifthRow = new JLabel("building direction by");       
107         container.add(fifthRow);        
108         JLabel sixthRow = new JLabel("pressing space key. Watch");  
109         container.add(sixthRow);        
110         JLabel seventhRow = new JLabel("out for the ball!"); 
111         container.add(seventhRow);
112     }
113     
114 }
