1   package pallopeli.graphics;
2   import java.awt.Color;
3   import java.awt.Graphics;
4   import pallopeli.objects.Ball;
5   
6   /**
7    * DrawablePiece provides draw-method for Ball.
8    * @author saara
9    */
10  
11  public class DrawableBall implements Drawable {
12      private Ball ballToDraw;
13  
14      public DrawableBall(Ball ballToDraw) {
15          this.ballToDraw = ballToDraw;
16      }
17      
18      
19  
20      @Override
21      public void draw(Graphics g) {
22          g.setColor(Color.RED);
23          g.fillOval(ballToDraw.getCurrentPosition().x, ballToDraw.getCurrentPosition().y, 2 * ballToDraw.getRadius(), 2 * ballToDraw.getRadius());
24      }
25      
26  }
