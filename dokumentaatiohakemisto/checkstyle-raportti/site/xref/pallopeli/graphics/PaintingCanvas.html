1   package pallopeli.graphics;
2   
3   import java.awt.Color;
4   import java.awt.Graphics;
5   import java.awt.List;
6   import java.util.ArrayList;
7   import javax.swing.JPanel;
8   import pallopeli.logic.Game;
9   
10  /**
11   * PaintingCanvas represents the background where all the drawable components are painted.
12   * @author saara
13   */
14  
15  public class PaintingCanvas extends JPanel implements Updateable {
16      private Game game;
17      private ArrayList<Drawable> drawables;
18      
19      public PaintingCanvas(Game game) {
20          super.setBackground(Color.WHITE);        
21          this.game = game;
22          this.drawables = this.createDrawables(game);
23      }
24      
25      /**
26       * Method inherited from JPanel; Calls the UI delegate's paint method, if the UI delegate is non-null.
27       * @param g 
28       */
29      @Override
30      protected void paintComponent(Graphics g) {
31          super.paintComponent(g);
32          for (int i = 0; i < this.drawables.size(); i++) {
33              this.drawables.get(i).draw(g);
34          }
35          
36      }
37      @Override
38      public void update() {
39          this.repaint();
40      }  
41      /**
42       * Creates the Drawable components of the given game.
43       * @param game
44       * @return List containing DrawableBoard and DrawableBall.
45       */
46      protected ArrayList<Drawable> createDrawables(Game game) {
47          ArrayList<Drawable> drawables = new ArrayList<>();
48          drawables.add(new DrawableBoard(game.getBoard()));
49          drawables.add(new DrawableBall(game.getBall()));
50          return drawables;
51      }
52      
53  }
